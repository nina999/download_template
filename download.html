<!DOCTYPE html>

<html>

<head>

    <title>自动下载Word文件</title>

    <script>

        // 使用GitHub Raw链接

        const fileUrl = "https://nina999.github.io/template_file/";

        

        // 自动下载函数

        function startDownload() {

            const link = document.createElement('a');

            link.href = fileUrl;

            link.download = 'document.docx'; // 下载的文件名

            document.body.appendChild(link);

            link.click();

            document.body.removeChild(link);

        }

        

        // 页面加载完成后触发下载

        window.onload = startDownload;

    </script>

</head>

<body>

    <p>如果下载未自动开始，请 <a href="#" onclick="startDownload()">点击这里</a>。</p>

</body>

</html>
